proc import datafile="C:\Users\User\Desktop\Personal Files\INNOVATION_STATION\Coursework\CSC423 DATA ANALYSIS\FINAL PROJECT\FOREIGN_GRAIN_DATA_2005.txt" 
out=GRAIN_DATA
dbms=DLM replace;
 delimiter=',';
*SHEET="sheet1";
*RANGE="Sheet1$A1:A237";
getnames=yes;
run;


/*SPLIT DATA*/
proc surveyselect data=GRAIN_DATA method=srs seed=49143 outall
  samprate=0.3 out=GRAIN_DATA_TST_TRN;
  run;

  data GRAIN_DATA_TRN;
  set GRAIN_DATA_TST_TRN(where=(Selected=0));
  run;

   data GRAIN_DATA_TST;
  set GRAIN_DATA_TST_TRN(where=(Selected=1));
  run;

/*exploring data*/
proc means data=GRAIN_DATA_TRN;
run;

proc univariate data=GRAIN_DATA_TRN;
run;


proc reg data=GRAIN_DATA_TRN;
  model SUPPLY_PRODUCTION  = YEAR SUPPLY_BEG_STOCKS SUPPLY_IMPORTS SUPPLY_TOTAL DSPRNCE_FD_ALCH_INDUS DSPRNCE_FEEDUSE DSPRNCE_DOMESTIC DSPRNCE_EXPORTS DSPRNCE_TOTALS STOCK_ENDINGS;
  plot r.*p. r.*nqq.;
   run;
   quit;

   proc reg data=GRAIN_DATA_TST;
  model SUPPLY_PRODUCTION  = YEAR SUPPLY_BEG_STOCKS SUPPLY_IMPORTS SUPPLY_TOTAL DSPRNCE_FD_ALCH_INDUS DSPRNCE_FEEDUSE DSPRNCE_DOMESTIC DSPRNCE_EXPORTS DSPRNCE_TOTALS STOCK_ENDINGS;
   plot r.*p. r.*nqq.;
   run;
   quit;

   proc rsquare adjrsq data=GRAIN_DATA_TRN;
    model SUPPLY_PRODUCTION  = YEAR SUPPLY_BEG_STOCKS SUPPLY_IMPORTS SUPPLY_TOTAL DSPRNCE_FD_ALCH_INDUS DSPRNCE_FEEDUSE DSPRNCE_DOMESTIC DSPRNCE_EXPORTS DSPRNCE_TOTALS STOCK_ENDINGS;
   run;
   quit;

   proc plot data=GRAIN_DATA_TRN; 
title 'Pairwise plot of FOREGRAIN DATA ';
plot SUPPLY_PRODUCTION*YEAR;
plot SUPPLY_PRODUCTION*SUPPLY_BEG_STOCKS;
plot SUPPLY_PRODUCTION*SUPPLY_IMPORTS;
plot SUPPLY_PRODUCTION*SUPPLY_TOTAL;
plot SUPPLY_PRODUCTION*DSPRNCE_FD_ALCH_INDUS;
plot SUPPLY_PRODUCTION*DSPRNCE_FEEDUSE;
plot SUPPLY_PRODUCTION*DSPRNCE_DOMESTIC;
plot SUPPLY_PRODUCTION*DSPRNCE_EXPORTS;
plot SUPPLY_PRODUCTION*DSPRNCE_TOTALS;
plot SUPPLY_PRODUCTION*STOCK_ENDINGS;
run;
